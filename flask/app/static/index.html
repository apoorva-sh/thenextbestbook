def compute_beta_at_j(X_train,Y_train,j,beta,lamda):
    n = X_train.shape[0]
    d = X_train.shape[1]
    X_minus_j = np.delete(X_train, j, axis=1)
    beta_minus_j = np.delete(beta, j, axis=0)
    
    R_minus_j = Y_train - (X_minus_j.dot(beta_minus_j))
    
    X_j = X_train[:,j].reshape(1, n)
    z_j = np.square(np.linalg.norm(X_j, ord=2))
    
    c_j = (2 / n) * X_j.dot(R_minus_j)
    denominator = (2/n )*z_j
    
    if c_j <= -lamda:
        beta_j = (lamda + c_j) / denominator
        return beta_j
    elif c_j >= lamda:
        beta_j = (-lamda + c_j) / denominator
        return beta_j
    elif np.abs(c_j) <= lamda:
        beta_j = 0
        return beta_j

def compute_obj(X_train,Y_train,beta,lamda):
    emp_risk = (1/X_train.shape[0])*np.square(np.linalg.norm(Y_train - X_train.dot(beta), ord=2))
    reg_param = lamda * np.linalg.norm(beta, ord=1)
    return emp_risk + reg_param

Write a function cycliccoorddescent that implements the cyclic coordinate descent algorithm. The cyclic coordinate descent algorithm proceeds sequentially. At each iteration, the algorithm increments the index jt of the coordinate to minimize over. Then the algorithm performs partial minimization with respect to the coordinate βjt cor- responding to that index. After updating the coordinate βjt, the algorithm proceeds to the next iteration. The function takes as input the initial point and the maximum number of iterations. The stopping criterion is the maximum number of iterations.¶

In [1947]:
def cycliccoorddescent(X_train,Y_train,lamda,maxIter):
    beta = np.zeros((X_train.shape[1], 1))
    d = X_train.shape[1]
    beta_history = []
    beta_history= np.array(beta)
    cost_history = []
    for i in range(0,maxIter):
        #print(beta)
        j = i % X_train.shape[1]
        beta[j, 0]= compute_beta_at_j(X_train,Y_train,j,beta,lamda)
        if i%d == 0:
            beta_history= np.append(beta_history,beta,axis=1)
            cost_history.append(compute_obj(X_train,Y_train,beta,lamda))
    return beta,beta_history,cost_history

    d = X_train.shape[1]
    beta_history = []

def normalized_oja(X ,eta, maxIter):
    n = X.shape[1]
    w_0 = np.ones((n,1))
    w_0 = w_0/np.sqrt(n)
    #print(np.linalg.norm(w_0,ord=2))
    for i in range(maxIter):
        w_t = w_0 + eta*np.dot(np.dot(X.T,X),w_0)
        w_t = w_t/np.linalg.norm(w_t,ord=2)
        w_0 = w_t        
    return w_t

def get_eigen_value(X,w):
    cov = np.cov(X.T)
    eigen_value = (w.T.dot(np.dot(cov,w)))/w.T.dot(w)  
    return eigen_value
